## Descripción

Obtiene información detallada de un juego específico usando su slug único.

## Path Parameters

<ParamField path="slug" type="string" required>
  Identificador único del juego (ej: "doom", "duke3d", "tetris")
</ParamField>

## Response

<ResponseField name="id" type="number">
  ID único del juego
</ResponseField>

<ResponseField name="name" type="string">
  Nombre del juego
</ResponseField>

<ResponseField name="slug" type="string">
  Identificador URL-friendly
</ResponseField>

<ResponseField name="description" type="string">
  Descripción completa del juego
</ResponseField>

<ResponseField name="year" type="number">
  Año de lanzamiento
</ResponseField>

<ResponseField name="publisher" type="string">
  Empresa publicadora
</ResponseField>

<ResponseField name="developer" type="string">
  Desarrollador del juego
</ResponseField>

<ResponseField name="genre" type="string">
  Género del juego
</ResponseField>

<ResponseField name="cover_url" type="string">
  URL de la imagen de portada (alta resolución)
</ResponseField>

<ResponseField name="screenshots" type="array">
  Array de URLs de capturas de pantalla
</ResponseField>

<ResponseField name="bundle_url" type="string">
  URL del archivo .jsdos del juego
</ResponseField>

<ResponseField name="players" type="number">
  Número de jugadores soportados
</ResponseField>

<ResponseField name="size_mb" type="number">
  Tamaño del bundle en megabytes
</ResponseField>

<ResponseField name="dosbox_config" type="object">
  Configuración recomendada de DOSBox

  <Expandable title="properties">
    <ResponseField name="cycles" type="string">
      Velocidad de CPU (ej: "max", "10000")
    </ResponseField>

    <ResponseField name="machine" type="string">
      Tipo de máquina (ej: "svga_s3", "vga")
    </ResponseField>

    <ResponseField name="core" type="string">
      Core de CPU ("dynamic", "normal")
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="controls" type="object">
  Información sobre controles del juego
</ResponseField>

<ResponseField name="highscore_count" type="number">
  Número total de puntuaciones registradas
</ResponseField>

<ResponseField name="created_at" type="string">
  Fecha de agregación al catálogo (ISO 8601)
</ResponseField>

## Ejemplo de Implementación

```javascript
// Cargar y lanzar juego

class GameLoader {
  async loadGame(slug) {
    try {
      // 1. Obtener metadata del juego
      const response = await fetch(`https://api.retrogamehub.games/games/${slug}`);

      if (!response.ok) {
        throw new Error('Juego no encontrado');
      }

      const game = await response.json();

      // 2. Mostrar información
      this.displayGameInfo(game);

      // 3. Iniciar descarga del bundle
      await this.downloadBundle(game.bundle_url, game.size_mb);

      // 4. Configurar DOSBox
      await this.configureDosBox(game.dosbox_config);

      // 5. Lanzar emulador
      await this.launchEmulator(game);

    } catch (error) {
      console.error('Error cargando juego:', error);
      this.showError(error.message);
    }
  }

  displayGameInfo(game) {
    document.getElementById('game-title').textContent = game.name;
    document.getElementById('game-year').textContent = game.year;
    document.getElementById('game-publisher').textContent = game.publisher;
    document.getElementById('game-description').textContent = game.description;

    // Mostrar controles
    const controlsDiv = document.getElementById('controls');
    Object.entries(game.controls).forEach(([action, key]) => {
      controlsDiv.innerHTML += `<p><strong>${action}:</strong> ${key}</p>`;
    });
  }

  async downloadBundle(url, sizeMB) {
    const progressBar = document.getElementById('progress');

    const response = await fetch(url);
    const reader = response.body.getReader();
    const contentLength = sizeMB * 1024 * 1024;

    let receivedLength = 0;
    const chunks = [];

    while(true) {
      const {done, value} = await reader.read();

      if (done) break;

      chunks.push(value);
      receivedLength += value.length;

      // Actualizar barra de progreso
      const percent = (receivedLength / contentLength) * 100;
      progressBar.style.width = `${percent}%`;
      progressBar.textContent = `${Math.round(percent)}%`;
    }

    // Combinar chunks
    const bundleData = new Uint8Array(receivedLength);
    let position = 0;
    for(let chunk of chunks) {
      bundleData.set(chunk, position);
      position += chunk.length;
    }

    return bundleData;
  }

  async configureDosBox(config) {
    // Aplicar configuración recomendada
    window.dosboxConfig = {
      wdosboxUrl: "/jsdos/wdosbox.js",
      ...config
    };
  }

  async launchEmulator(game) {
    // Iniciar js-dos con el bundle
    const dos = await Dos(document.getElementById('canvas-container'), window.dosboxConfig);

    dos.run(game.bundle_url);

    // Fullscreen automático
    requestFullscreen();

    // Iniciar score tracking
    scoreTracker.start(game.slug);
  }

  showError(message) {
    const errorDiv = document.createElement('div');
    errorDiv.className = 'error-modal';
    errorDiv.innerHTML = `
      <h2>Error</h2>
      <p>${message}</p>
      <button onclick="this.parentElement.remove()">OK</button>
    `;
    document.body.appendChild(errorDiv);
  }
}

// Uso
const loader = new GameLoader();
const gameSlug = new URLSearchParams(window.location.search).get('game');
loader.loadGame(gameSlug);

```

## Caching

<Info>
  Los detalles de juegos se cachean en CloudFront por **24 horas**. Usa el header `If-None-Match` para validación de caché.
</Info>

## Rate Limiting

- **Límite**: 100 peticiones por minuto por IP

- **Uso típico**: 1 petición al cargar la página del juego
