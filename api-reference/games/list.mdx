## Descripción

Obtiene el catálogo completo de juegos disponibles con su metadata, incluyendo nombre, descripción, año, género y URLs de assets.

## Query Parameters

<ParamField query="genre" type="string">
  Filtrar por género: `fps`, `platformer`, `rpg`, `puzzle`, `strategy`, `fighting`
</ParamField>

<ParamField query="year" type="number">
  Filtrar por año de lanzamiento (ej: 1993)
</ParamField>

<ParamField query="search" type="string">
  Búsqueda por nombre o descripción (case-insensitive)
</ParamField>

<ParamField query="limit" type="number" default="50">
  Número máximo de resultados (1-100)
</ParamField>

<ParamField query="offset" type="number" default="0">
  Offset para paginación
</ParamField>

## Response

<ResponseField name="games" type="array">
  Array de objetos de juego

  <Expandable title="Game Object">

      ID único del juego
    </ResponseField>

    <ResponseField name="name" type="string">

    </ResponseField>

    <ResponseField name="slug" type="string">

    <ResponseField name="description" type="string">

## Paginación

```javascript
// Cargar primera página (10 juegos)
let offset = 0;
const limit = 10;

async function loadPage() {
  const response = await fetch(
    `https://api.retrogamehub.games/games?limit=${limit}&offset=${offset}`
  );

  const data = await response.json();

  renderGames(data.games);

  // Habilitar botón "Siguiente" si hay más
  if (offset + limit < data.total) {
    enableNextButton();
  }
}

function nextPage() {
  offset += limit;
  loadPage();
}

function previousPage() {
  offset = Math.max(0, offset - limit);
  loadPage();
}

```

## Caching

<Info>
  Los resultados de esta API se cachean en CloudFront por **1 hora**. Los headers `Cache-Control` indican el tiempo de caché restante.
</Info>

## Rate Limiting

- **Límite**: 100 peticiones por minuto por IP

- **Header**: `X-RateLimit-Remaining`

- **Recomendación**: Cachea los resultados en el frontend

## Ejemplo de Uso Completo

```javascript
// games.html - Cargar y renderizar catálogo

class GameCatalog {
  constructor() {
    this.games = [];
    this.currentGenre = null;
  }

  async loadGames(genre = null) {
    const url = new URL('https://api.retrogamehub.games/games');

    if (genre) {
      url.searchParams.set('genre', genre);
    }

    const response = await fetch(url);
    const data = await response.json();

    this.games = data.games;
    this.renderGrid();
  }

  renderGrid() {
    const grid = document.getElementById('games-grid');
    grid.innerHTML = '';

    this.games.forEach(game => {
      const card = document.createElement('div');
      card.className = 'game-card';
      card.innerHTML = `
        <img src="${game.cover_url}" alt="${game.name}">
        <h3>${game.name}</h3>
        <p>${game.year} - ${game.publisher}</p>
        <button onclick="playGame('${game.slug}')">
          JUGAR
        </button>
      `;

      grid.appendChild(card);
    });
  }

  filterByGenre(genre) {
    this.currentGenre = genre;
    this.loadGames(genre);
  }
}

// Inicializar
const catalog = new GameCatalog();
catalog.loadGames();

```
