## Ejemplo de Perfil de Usuario

```javascript
// Renderizar perfil completo del usuario

class UserProfile {
  constructor(userId) {
    this.userId = userId;
    this.data = null;
  }

  async load() {
    const response = await fetch(
      `https://api.retrogamehub.games/scores/user/${this.userId}`
    );

    this.data = await response.json();
    this.render();
  }

  render() {
    const container = document.getElementById('profile-container');

    container.innerHTML = `
      <div class="profile-header">
        <h1>${this.data.user.display_name}</h1>
        <p>@${this.data.user.username}</p>
      </div>

      <div class="profile-stats">
        <div class="stat">
          <span class="stat-value">${this.data.total_games_played}</span>
          <span class="stat-label">Juegos Jugados</span>
        </div>

        <div class="stat">
          <span class="stat-value">#${this.data.average_rank}</span>
          <span class="stat-label">Ranking Promedio</span>
        </div>

        <div class="stat">
          <span class="stat-value">${this.getBestRank()}</span>
          <span class="stat-label">Mejor Ranking</span>
        </div>
      </div>

      <div class="scores-list">
        <h2>Puntuaciones</h2>
        ${this.renderScoresList()}
      </div>
    `;
  }

  renderScoresList() {
    return this.data.scores
      .sort((a, b) => a.rank - b.rank) // Ordenar por mejor ranking
      .map(score => `
        <div class="score-card">
          <div class="score-game">${score.game_name}</div>
          <div class="score-value">${score.score.toLocaleString()}</div>
          <div class="score-rank">
            Ranking: #${score.rank} de ${score.total_players}
          </div>
          <div class="score-date">
            √öltimo: ${new Date(score.updated_at).toLocaleDateString()}
          </div>
        </div>
      `)
      .join('');
  }

  getBestRank() {
    if (this.data.scores.length === 0) return '-';

    const best = Math.min(...this.data.scores.map(s => s.rank));
    return `#${best}`;
  }
}

// Uso
const profile = new UserProfile(123);
profile.load();

```
