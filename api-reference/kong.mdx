## docker-publish-and-update-k8s.yml
```yaml
name: Compilar, Publicar y Actualizar Kubernetes

on:
  push:
    branches:
      - main
    paths-ignore:
      - '**.md'
      - '.gitignore'

env:
  REGISTRY_GHCR: ghcr.io
  REGISTRY_DOCKERHUB: docker.io
  IMAGE_NAME: retrogamecloud/kong
  K8S_REPO: retrogamecloud/kubernetes
  K8S_MANIFEST_FILE: 05-kong.yaml

jobs:
  test:
    runs-on: ubuntu-latest
    
    steps:
      - name: Clonar repositorio
        uses: actions/checkout@v4

      - name: Configurar Node.js
        uses: actions/setup-node@v4
        with:
          node-version: '20'
          cache: 'npm'

      - name: Instalar dependencias
        run: npm ci

      - name: Validar configuraci√≥n de Kong
        run: npm test

  build-and-push:
    needs: test
    runs-on: ubuntu-latest
    permissions:
      contents: read
      packages: write

    steps:
      - name: Clonar repositorio
        uses: actions/checkout@v4
        with:
          fetch-depth: 0

      - name: Configurar Docker Buildx
        uses: docker/setup-buildx-action@v3

      - name: Iniciar sesi√≥n en GitHub Container Registry
        uses: docker/login-action@v3
        with:
          registry: ${{ env.REGISTRY_GHCR }}
          username: ${{ github.actor }}
          password: ${{ secrets.GITHUB_TOKEN }}

      - name: Iniciar sesi√≥n en Docker Hub
        uses: docker/login-action@v3
        with:
          registry: ${{ env.REGISTRY_DOCKERHUB }}
          username: ${{ secrets.DOCKERHUB_USERNAME }}
          password: ${{ secrets.DOCKERHUB_TOKEN }}

      - name: Generar etiqueta de versi√≥n y SHA
        id: version
        run: |
          # Contar commits desde el inicio
          COMMIT_COUNT=$(git rev-list --count HEAD)
          VERSION="v1.0.${COMMIT_COUNT}"
          SHORT_SHA=$(git rev-parse --short HEAD)
          
          echo "version=${VERSION}" >> $GITHUB_OUTPUT
          echo "short_sha=${SHORT_SHA}" >> $GITHUB_OUTPUT
          echo "Versi√≥n generada: ${VERSION}"
          echo "SHA corto: ${SHORT_SHA}"

      - name: Extraer metadatos
        id: meta
        uses: docker/metadata-action@v5
        with:
          images: |
            ghcr.io/retrogamecloud/kong
            retrogamehub/kong
          tags: |
            type=raw,value=latest
            type=raw,value=${{ steps.version.outputs.version }}
            type=sha,prefix=sha-,format=short

      - name: Compilar y publicar imagen Docker
        uses: docker/build-push-action@v5
        with:
          context: .
          push: true
          tags: ${{ steps.meta.outputs.tags }}
          labels: ${{ steps.meta.outputs.labels }}
          cache-from: type=registry,ref=${{ env.REGISTRY_GHCR }}/${{ env.IMAGE_NAME }}:buildcache
          cache-to: type=registry,ref=${{ env.REGISTRY_GHCR }}/${{ env.IMAGE_NAME }}:buildcache,mode=max
          platforms: linux/amd64

      - name: Resumen de imagen publicada
        run: |
          echo "‚úÖ Imagen publicada exitosamente"
          echo "üì¶ Versi√≥n: ${{ steps.version.outputs.version }}"
          echo "üîñ SHA: sha-${{ steps.version.outputs.short_sha }}"

  update-kubernetes-manifest:
    needs: build-and-push
    runs-on: ubuntu-latest
    permissions:
      contents: write

    steps:
      - name: Clonar repositorio de Kubernetes
        uses: actions/checkout@v4
        with:
          repository: ${{ env.K8S_REPO }}
          token: ${{ secrets.K8S_UPDATE_TOKEN }}
          ref: main

      - name: Obtener informaci√≥n de la imagen desde el job anterior
        id: image_info
        run: |
          # Reutilizar el SHA del commit actual sin clonar de nuevo
          SHORT_SHA=$(echo "${{ github.sha }}" | cut -c1-7)
          COMMIT_COUNT=${{ github.run_number }}
          VERSION="v1.0.${COMMIT_COUNT}"
          
          echo "version=${VERSION}" >> $GITHUB_OUTPUT
          echo "short_sha=${SHORT_SHA}" >> $GITHUB_OUTPUT
          echo "image_tag=sha-${SHORT_SHA}" >> $GITHUB_OUTPUT

      - name: Actualizar manifiesto de Kubernetes
        run: |
          IMAGE_TAG="${{ steps.image_info.outputs.image_tag }}"
          VERSION="${{ steps.image_info.outputs.version }}"
          
          # Actualizar la imagen en el manifiesto usando sed
          sed -i "s|image: retrogamehub/kong:.*|image: retrogamehub/kong:${IMAGE_TAG}|g" ${{ env.K8S_MANIFEST_FILE }}
          
          # Actualizar el comentario con la versi√≥n
          sed -i "s|# Imagen de Kong desde DockerHub.*|# Imagen de Kong desde DockerHub - ${VERSION} (${IMAGE_TAG})|g" ${{ env.K8S_MANIFEST_FILE }}
          
          echo "‚úÖ Manifiesto actualizado con imagen: ${IMAGE_TAG}"

      - name: Configurar Git
        run: |
          git config user.name "GitHub Actions Bot"
          git config user.email "actions@github.com"

      - name: Crear rama y PR con cambios
        run: |
          VERSION="${{ steps.image_info.outputs.version }}"
          IMAGE_TAG="${{ steps.image_info.outputs.image_tag }}"
          BRANCH_NAME="auto/kong-${IMAGE_TAG}"
          
          git add ${{ env.K8S_MANIFEST_FILE }}
          
          # Solo proceder si hay cambios
          if git diff --staged --quiet; then
            echo "No hay cambios en el manifiesto"
          else
            # Crear rama para el PR
            git checkout -b "${BRANCH_NAME}"
            
            git commit -m "chore: Actualizar imagen de Kong a ${IMAGE_TAG} [skip ci]" \
                       -m "Version: ${VERSION}" \
                       -m "SHA: ${{ steps.image_info.outputs.short_sha }}" \
                       -m "Commit origen: ${{ github.sha }}" \
                       -m "Actualizacion automatica desde CI/CD del repositorio kong"
            
            git push origin "${BRANCH_NAME}"
            
            # Crear PR usando GitHub CLI
            PR_URL=$(gh pr create \
              --repo ${{ env.K8S_REPO }} \
              --base main \
              --head "${BRANCH_NAME}" \
              --title "chore(kong): Actualizar a ${IMAGE_TAG}" \
              --body "Actualizacion automatica de Kong - Version: ${VERSION}, Tag: ${IMAGE_TAG}, SHA: ${{ steps.image_info.outputs.short_sha }}, Commit: ${{ github.sha }}")
            
            echo "‚úÖ PR creado: ${PR_URL}"
            
            # Auto-aprobar el PR
            gh pr review "${PR_URL}" --approve --body "Auto-aprobado por CI/CD"
            
            # Hacer merge autom√°tico
            gh pr merge "${PR_URL}" --auto --squash --delete-branch
            
            echo "‚úÖ PR auto-aprobado y configurado para merge autom√°tico"
          fi
        env:
          GH_TOKEN: ${{ secrets.K8S_UPDATE_TOKEN }}

      - name: Resumen
        run: |
          echo "## Actualizacion de Kubernetes Completada" >> $GITHUB_STEP_SUMMARY
          echo "" >> $GITHUB_STEP_SUMMARY
          echo "- **Version**: ${{ steps.image_info.outputs.version }}" >> $GITHUB_STEP_SUMMARY
          echo "- **Tag de imagen**: ${{ steps.image_info.outputs.image_tag }}" >> $GITHUB_STEP_SUMMARY
          echo "- **Manifiesto actualizado**: ${{ env.K8S_MANIFEST_FILE }}" >> $GITHUB_STEP_SUMMARY
          echo "- **Repositorio**: ${{ env.K8S_REPO }}" >> $GITHUB_STEP_SUMMARY
```
