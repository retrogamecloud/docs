## sync-backend-manifest.yml
```yaml
name: Sync Backend Manifest to Infrastructure

on:
  push:
    branches:
      - main
    paths:
      - '02-backend.yaml'

jobs:
  sync-to-infrastructure:
    runs-on: ubuntu-latest
    steps:
      - name: Checkout kubernetes repo
        uses: actions/checkout@v4
        with:
          path: kubernetes

      - name: Checkout infrastructure repo
        uses: actions/checkout@v4
        with:
          repository: retrogamecloud/infrastructure
          token: ${{ secrets.PAT_TOKEN }}
          path: infrastructure

      - name: Copy backend manifest
        run: |
          cp kubernetes/02-backend.yaml infrastructure/argocd/backend-overlays/base/02-backend.yaml

      - name: Commit and push changes
        run: |
          cd infrastructure
          git config user.name "github-actions[bot]"
          git config user.email "github-actions[bot]@users.noreply.github.com"
          
          # Crear rama con timestamp para evitar conflictos
          BRANCH_NAME="sync/backend-manifest-$(date +%s)"
          git checkout -b $BRANCH_NAME
          
          git add argocd/backend-overlays/base/02-backend.yaml
          
          # Solo commit y push si hay cambios
          if ! git diff --staged --quiet; then
            git commit -m "chore: sync backend manifest from kubernetes repo"
            git push origin $BRANCH_NAME
            
            # Crear PR usando GitHub CLI
            echo "Creating PR..."
            gh pr create \
              --repo retrogamecloud/infrastructure \
              --base main \
              --head $BRANCH_NAME \
              --title "chore: sync backend manifest from kubernetes repo" \
              --body "Automated sync of 02-backend.yaml from kubernetes repository" \
              --label "automated"
          else
            echo "No changes to sync"
          fi
        env:
          GH_TOKEN: ${{ secrets.PAT_TOKEN }}
```

  serviceName: "postgres-db"

  selector:

      app: postgres-db
  template:
    metadata:
      labels:
        app: postgres-db
    spec:
      containers:
        - name: postgres
          image: postgres:15-alpine
          ports:
            - containerPort: 5432
          envFrom:
            - secretRef:
                name: postgres-secret
          volumeMounts:
            - name: postgres-storage
              mountPath: /var/lib/postgresql/data
            - name: postgres-init-db
              mountPath: /docker-entrypoint-initdb.d
      volumes:
        - name: postgres-init-db
          configMap:
            name: postgres-init-cm

  # Definición del volumen persistente

    - metadata:
        name: postgres-storage
      spec:
        accessModes: ["ReadWriteOnce"]
        resources:
          requests:
            storage: 2Gi
